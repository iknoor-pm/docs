```python Python
import requests

API_KEY = "YOUR_API_KEY"

response = requests.get(
    "https://smartdelivery.smartlead.ai/api/v1/smart-delivery/blacklists",
    params={"api_key": API_KEY}
)

result = response.json()
print(result)
```
