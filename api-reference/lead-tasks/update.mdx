---
title: "Update Lead Task"
api: "PATCH https://server.smartlead.ai/api/v1/leads/{lead_id}/tasks/{task_id}"
description: "Update an existing task for a lead"
---

## Path Parameters

<ParamField path="lead_id" type="number" required>
  The unique identifier of the lead
</ParamField>

<ParamField path="task_id" type="number" required>
  The unique identifier of the task
</ParamField>

## Query Parameters

<ParamField query="api_key" type="string" required>
  Your SmartLead API key
</ParamField>

## Body Parameters

<ParamField body="title" type="string">
  Updated task title
</ParamField>

<ParamField body="description" type="string">
  Updated task description
</ParamField>

<ParamField body="due_date" type="string">
  Updated due date in ISO 8601 format
</ParamField>

<ParamField body="status" type="string">
  Updated status. Values: `PENDING`, `COMPLETED`
</ParamField>

<RequestExample>

```bash cURL
curl -X PATCH "https://server.smartlead.ai/api/v1/leads/123/tasks/789?api_key=YOUR_KEY" \
  -H "Content-Type: application/json" \
  -d '{"status": "COMPLETED"}'
```

</RequestExample>

## Response Fields

<ResponseField name="ok" type="boolean">
  Whether the operation was successful
</ResponseField>

<ResponseField name="data" type="object">
  Updated task object
</ResponseField>

<ResponseExample>

{
